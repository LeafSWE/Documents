	\subsection{SDK}
		Framework di sviluppo per applicazioni Android.
		\begin{itemize} 
			\item Vesione SDK 24.4.1;
			\item Versione build tools 23.0.3;
			\item Versione target SDK 23;
			\item Versione minima SDK 19.
		\end{itemize}

		\end{itemize}

			\item Vesione oraclejdk8.
		\end{itemize}
